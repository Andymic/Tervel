161  template<class T>
162  T Descriptor::remove(T t, std::atomic<T> *address) {
163  if (rDepth > thread::nThreads+1) {
164  rReturn = true;
165  return reinterpret_cast<T>(nullptr); // result not used
166  }
167  rDepth++;
168 
169  Descriptor *descr = Descriptor::unmark(t);
170 
171  bool watched = PoolElem::watch(descr,
172  reinterpret_cast<std::atomic<void *>*>(address),
173  reinterpret_cast<void *>(t));
174  T newValue;
175  if (watched) {
176  newValue = reinterpret_cast<T>(descr->descr_complete(
177  reinterpret_cast<void *>(t),
178  reinterpret_cast<std::atomic<void *>*>(address)));
179 
180  PoolElem::unwatch(descr);
181  } else {
182  newValue = address->load();
183  }
184 
185  rDepth--;
186  return newValue;
187  }
